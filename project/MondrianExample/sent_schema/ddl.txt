
CREATE DATABASE sent_schema;

CREATE TABLE fact_sentence (
  `id` bigint(20) DEFAULT NULL,
  `place_id` bigint(20) DEFAULT NULL,
  `sentence_id` bigint(20) DEFAULT NULL,
  `frequency` bigint(20) DEFAULT NULL
);

DROP TABLE IF EXISTS place;
CREATE TABLE place
         (
           place_id     INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
           place_name   VARCHAR(200)
         );
CREATE UNIQUE INDEX idx_place_pk ON place
(
    place_id
);

DROP TABLE IF EXISTS sentence;
CREATE TABLE sentence
(
    sentence_id  BIGINT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    book_name    VARCHAR(5000),
    sentence     VARCHAR(5000)
);
